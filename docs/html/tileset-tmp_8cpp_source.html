79 void Tilesettmp::loadTiles(string filename, int iw, int ih) {
80  ifstream in(filename.c_str());
81  if(!in.is_open()) {
82  cout << "Problem with loading the file" << endl;
83  return;
84  }
85  int w, h;
86  in >> w;
87  in >> h;
88  int current;
89  layer cur;
90  cur.tw = iw;
91  cur.th = ih;
92  cur.width = w;
93  cur.height = h;
94  for(int i=0;i<h;i++) {
95  for(int j=0;j<w;j++) {
96  if(in.eof()) {
97  cout << "File end reached too soon" << endl;
98  return;
99  }
100  in >> current;
101  if(current != 0) {
102  tile tmp;
103  tmp.tile = tiles[current];
104  tmp.x=j;
105  tmp.y=i;
106  cur.tiles.push_back(tmp);
107  }
108  }
109  }
110  if(!in.eof()) {
111  int setX, setY;
112  in >> setX;
113  in >> setY;
114  cur.x = setX;
115  cur.y = setY;
116  if(!set) {
117  setCoord(-cur.x, -cur.y);
118  } else {
119  setCoord(0, 0);
120  }
121  }
122  in.close();
123  tileset.push_back(cur);
124  setCoord(-cur.x, -cur.y);
125  move(-cur.x, -cur.y);
126 }
