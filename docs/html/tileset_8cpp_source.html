78 void Tileset::loadTiles(string filename, int iw, int ih) {
79  ifstream in(filename.c_str());
80  if(!in.is_open()) {
81  cout << "Problem with loading the file" << endl;

83  }
84  int w, h;

87  int current;
88  layer cur;
89  cur.tw = iw;

92  cur.height = h;
93  for(int i=0;i<h;i++) {
94  for(int j=0;j<w;j++) {
95  if(in.eof()) {
96  cout << "File end reached too soon" << endl;
97  return;
98  }
99  in >> current;
100  if(current != 0) {
101  tile tmp;
102  tmp.tile = tiles[current];
103  tmp.x=j;
104  tmp.y=i;
105  cur.tiles.push_back(tmp);
106  }
107  }
108  }
109  if(!in.eof()) {
110  int setX, setY;
111  in >> setX;
112  in >> setY;
113  cur.x = setX;
114  cur.y = setY;
115  if(!set) {
116  setCoord(-cur.x, -cur.y);
117  } else {
118  setCoord(0, 0);
119  }
120  }
121  in.close();
122  tileset.push_back(cur);
123  setCoord(-cur.x, -cur.y);
124  move(-cur.x, -cur.y);
